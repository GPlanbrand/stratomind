model GuestSession {
  id        String   @id @default(cuid())
  token     String   @unique

  expireAt  DateTime
  createdAt DateTime @default(now())

```prisma

  used      Boolean  @default(false)
  expireAt  DateTime
  createdAt DateTime @default(now())
  
  @@index([phone, createdAt])
  @@map("sms_codes")
}
```
